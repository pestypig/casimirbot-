4b) make a small “compare” wrapper

Create WarpBubbleCompare.tsx:

import React from 'react';
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card';
import { WarpVisualizer } from './WarpVisualizer';

export default function WarpBubbleCompare({ parameters }: { parameters: any }) {
  const base = parameters || {};
  const leftReal = {
    ...base,
    physicsParityMode: true,               // hard parity
    viz: {
      colorMode: 'shear',                  // show σ in Real view
      curvatureGainT: 0,                   // unity
      curvatureBoostMax: 1,
      exposure: 3.0, zeroStop: 1e-5,       // low-contrast, honest mapping
      cosmeticLevel: 1
    }
  };
  const rightShow = {
    ...base,
    physicsParityMode: false,              // cosmetics on
    viz: {
      colorMode: 'theta',                  // θ diverging for instant front/back
      curvatureGainT: 0.50,                // demo mid
      curvatureBoostMax: 40,
      exposure: 6.0, zeroStop: 1e-7,
      cosmeticLevel: 10
    }
  };

  return (
    <Card className="h-full">
      <CardHeader className="pb-2">
        <CardTitle className="text-lg">Real vs Show — Natário Bubble</CardTitle>
      </CardHeader>
      <CardContent>
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-4">
          <div className="rounded-lg border border-slate-700 overflow-hidden">
            <div className="px-3 py-2 text-xs font-mono bg-slate-900/80 text-sky-300">
              REAL (Parity) • σ coloring • honest scale
            </div>
            <div style={{height:'min(56vh,520px)'}}>
              <WarpVisualizer parameters={leftReal}/>
            </div>
          </div>
          <div className="rounded-lg border border-slate-700 overflow-hidden">
            <div className="px-3 py-2 text-xs font-mono bg-slate-900/80 text-amber-300">
              SHOW (Hero) • θ coloring • boosted scale
            </div>
            <div style={{height:'min(56vh,520px)'}}>
              <WarpVisualizer parameters={rightShow}/>
            </div>
          </div>
        </div>
      </CardContent>
    </Card>
  );
}
