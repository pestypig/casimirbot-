+const fmtExp = (v: unknown, digits = 3) =>

+function stats(arr: ArrayLike<number>) {
+  let min = Infinity, max = -Infinity, sum = 0;
+  const N = arr.length;
+  for (let i = 0; i < N; i++) {
+    const v = Number(arr[i]);
+    if (!Number.isFinite(v)) continue;
+    if (v < min) min = v;
+    if (v > max) max = v;
+    sum += v;
+  }
+  return { min, max, mean: N ? sum / N : NaN, N };
+}
