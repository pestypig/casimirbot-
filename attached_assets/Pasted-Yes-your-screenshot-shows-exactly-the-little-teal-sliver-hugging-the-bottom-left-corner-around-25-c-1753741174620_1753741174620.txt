import { viability }       from 'sim_core/viability';
import { PipelineParams, ConstraintConfig } from 'sim_core/types';

// 1) Re-declare the exact preset constants
const pipe: PipelineParams = {
  gap:       1e-9,
  A_tile:    25e-4,          // 25 cm² → m²
  gamma_geo: 25,
  Q:         1e9,
  duty:      0.01,
  duty_eff:  0.01 / 400,
  N_tiles:   1.96e9,
  P_raw:     2e15
};

const cons: ConstraintConfig = {
  massNominal: 1400,
  massTolPct:  5,
  maxPower:    100,  // MW
  maxZeta:     1.0,
  minGamma:    25
};

// 2) Sanity check: the preset point must be viable
console.assert(
  viability(25, 5.0, pipe, cons).ok,
  'Needle–Hull preset should be viable'
);

// 3) Check a small patch around it
for (let da of [20, 25, 30]) {
  for (let R of [4.0, 5.0, 6.0]) {
    const { ok, m_exotic, P_avg, zeta } = viability(da, R, pipe, cons);

  }
}
