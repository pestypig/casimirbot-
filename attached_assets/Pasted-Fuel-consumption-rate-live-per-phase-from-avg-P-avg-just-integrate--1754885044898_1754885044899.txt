Below is a compact patch that adds a trip telemetry HUD to the player. It records energy, time, and distance per mode and shows a summary when the run completes.

// TripPlayer.tsx â€” add near top:
type Telemetry = {
  energyJ: number;              // total J
  distanceLy: number;           // total ly
  byMode: Record<string, { tSec: number; energyJ: number; distanceLy: number }>;
};

// inside TripPlayer component:
const [tele, setTele] = React.useState<Telemetry>({ energyJ:0, distanceLy:0, byMode:{} });

// helper: effective ly/hour (use your existing computeEffectiveLyPerHour or inject one)
function vEff(mode:string){ /* same function you used in FuelGauge */ 
  return computeEffectiveLyPerHour(mode, /* duty etc if you have them */);
}

// in the effect â€œstep()â€ tick, right after you read `const s = getState();`:
const dt = 0.5; // seconds per tick
const curMode = (document.querySelector("[data-mode-name]")?.textContent ?? "") || phase; // or pass mode as prop
const pW = (s.powerMW ?? 0) * 1e6;                 // MW -> W
const vLyPerHour = vEff(curMode);
const dLy = vLyPerHour * (dt / 3600);              // ly this tick
const dE  = pW * dt;                               // J this tick

setTele(prev => {
  const m = prev.byMode[curMode] ?? { tSec:0, energyJ:0, distanceLy:0 };
  m.tSec += dt; m.energyJ += dE; m.distanceLy += dLy;
  return {
    energyJ: prev.energyJ + dE,
    distanceLy: prev.distanceLy + dLy,
    byMode: { ...prev.byMode, [curMode]: m },
  };
});
